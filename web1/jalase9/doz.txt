اول باید یک آرایه دوبعدی در کد جاوا اسکریپت تعریف کنی
let map = [];
let row = 3;
let col = 3;
let h = 0
 
// Loop to initialize 2D array elements.
for (let i = 0; i < row; i++) {
    map[i] = [];
    for (let j = 0; j < col; j++) {
        map[i][j] = h++;
    }
}
console.log(map);
